CREATE TRIGGER tr1 BEFORE INSERT ON nbase_theme
FOR EACH ROW
BEGIN
IF(SELECT COUNT(ID_Theme) FROM nbase_theme)>9
THEN 
SIGNAL SQLSTATE '45000' 
SET MESSAGE_TEXT = 'Themes less than 10.';
END IF;
END $$
